# Agregar valores encima de cada barra
for barra in barras:
    altura = barra.get_height()
    plt.text(barra.get_x() + barra.get_width()/2, altura,
        f'{int(altura):,}',
        ha='center', va='bottom',
        fontsize=10#, fontweight='bold'
    )

plt.ylabel('Cantidad')
plt.title('Ajuses en periodo')

# Mostrar únicamente 0 y 1 en eje X
plt.xticks(
    ticks=range(len(tipo)),
    labels=tipo.index.astype(str))

plt.grid(axis='y', alpha=0.3)
plt.show()

# Conteo de categorías
proporcion = (df['AJUSTE_PERIODO'].value_counts().sort_index().reset_index())

proporcion.columns = ['Ajuste', 'Cantidad']

# Etiquetas descriptivas
proporcion['Ajuste'] = proporcion['Ajuste'].map({
    0: 'Sin ajuste',
    1: 'Con ajuste'})

# Gráfico interactivo
fig = px.pie(
    proporcion,
    names='Ajuste', values='Cantidad',
    title='Proporción de ajustes',
    hole=0.45  # tipo dona
    )

fig.update_traces(
    textposition='inside', textinfo='percent+label',
    hovertemplate=
    '<b>%{label}</b><br>' +
    'Cantidad: %{value:,}<br>' +
    'Porcentaje: %{percent}<extra></extra>')

fig.update_layout(template='plotly_white', width=600, height=500,
    legend_title='Estado')

fig.show()

variables_numericas = df.select_dtypes(include=['int64', 'float64', 'int32', 'float32']).columns

resultado = []
for col in variables_numericas:
    serie = df[col].dropna()

    minimo = serie.min() #Mínimos
    negativos = (serie < 0).sum() # Negativos

    # Outliers por IQR
    q1 = serie.quantile(0.25)
    q3 = serie.quantile(0.75)
    iqr = q3 - q1

    limite_inferior = q1 - 1.5 * iqr
    limite_superior = q3 + 1.5 * iqr

    outliers = ((serie < limite_inferior) |
                (serie > limite_superior)).sum()

    resultado.append({'Variable': col,
                      'Minimo': minimo,
                      'Negativos': negativos,
                      '% Negativos': round(100 * negativos / len(serie), 2),
                      'Outliers': outliers,
                      '% Outliers': round(100 * outliers / len(serie), 2)})

reporte_outliers = pd.DataFrame(resultado)

reporte_outliers = reporte_outliers.sort_values(
    ['% Outliers', '% Negativos'], ascending=False)
